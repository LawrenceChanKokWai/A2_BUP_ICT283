280 template<class T>
281 float AoAvlMap<T>::CalculateSPCC(const Vector<float> &dataOne, const Vector<float> &dataTwo ) const
282 {
283  float sumValueOne = 0.0;
284  float sumValueTwo = 0.0;
285 
286  // Check if the sizes of dataOne and dataTwo are equal
287  if (dataOne.GetUsed() != dataTwo.GetUsed() || dataOne.IsEmpty() || dataTwo.IsEmpty())
288  {
289  // Return an error value
290  return -1.0;
291  }
292 
293  for( unsigned i(0); i < dataOne.GetUsed(); i++ )
294  {
295  sumValueOne += dataOne[i];
296  }
297  for( unsigned i(0); i < dataTwo.GetUsed(); i++ )
298  {
299  sumValueTwo += dataTwo[i];
300  }
301 
302  // Calculate means of dataOne and dataTwo
303  float mean1 = sumValueOne / dataOne.GetUsed();
304  float mean2 = sumValueTwo / dataTwo.GetUsed();
305 
306  // Calculate covariance and variances
307  float cov = 0.0;
308  float valueOne = 0.0;
309  float valueTwo = 0.0;
310 
311  for (unsigned i(0); i < dataOne.GetUsed(); ++i)
312  {
313  cov += (dataOne[i] - mean1) * (dataTwo[i] - mean2);
314  valueOne += pow(dataOne[i] - mean1, 2);
315  valueTwo += pow(dataTwo[i] - mean2, 2);
316  }
317  cov /= dataOne.GetUsed();
318  valueOne /= dataOne.GetUsed();
319  valueTwo /= dataTwo.GetUsed();
320 
321  // Calculate Sample Pearson Correlation Coefficient (sPCC)
322  float spcc = cov / (sqrt(valueOne) * sqrt(valueTwo));
323 
324  return spcc;
325 }
